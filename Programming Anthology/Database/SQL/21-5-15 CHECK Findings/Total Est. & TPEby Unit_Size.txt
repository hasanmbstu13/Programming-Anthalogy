SELECT UNIT_SIZE,COUNT(*) AS EST
FROM BBSEC2013_REPORTS
WHERE  (TO_NUMBER(SUBSTR(Q6_IND_CODE_CLASS_CODE,1,2)) BETWEEN 05 AND 99) 
GROUP BY UNIT_SIZE
ORDER BY UNIT_SIZE;


SELECT UNIT_SIZE,SUM(TOTAL_PERSON_ENGAGED) AS TPE
FROM BBSEC2013_REPORTS
WHERE  (TO_NUMBER(SUBSTR(Q6_IND_CODE_CLASS_CODE,1,2)) BETWEEN 05 AND 99) 
GROUP BY UNIT_SIZE
ORDER BY UNIT_SIZE;
