--Basic Information of Economomic Census 2013 
--Query

--Total Est. In Questionaries
SELECT COUNT(*)
FROM QUESTIONAIRES;

--Total Est. In Reports
SELECT COUNT(*)
FROM BBSEC2013_REPORTS;

--Economic Household
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE Q4_UNIT_TYPE = 1;

--Establishment
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE Q4_UNIT_TYPE = 2;

--Permanent Est.
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE Q4_UNIT_TYPE = 2
AND Q4_UNIT_ORG_TYPE = 1;

--Temporary Est.
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE Q4_UNIT_TYPE = 2
AND Q4_UNIT_ORG_TYPE = 2;

--Govt. Est.
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE Q9_LEGAL_ENTITY_TYPE_ID = '05';

--Autonomous. Est.
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE Q9_LEGAL_ENTITY_TYPE_ID = '06';

--Total TPE
SELECT SUM(TOTAL_PERSON_ENGAGED)
FROM BBSEC2013_REPORTS;

--Total Male Engaged
SELECT SUM(TOTAL_MALE_ENGAGED)
FROM BBSEC2013_REPORTS;

--Total Female Engaged
SELECT SUM(TOTAL_FEMALE_ENGAGED)
FROM BBSEC2013_REPORTS;

--Rural
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE QUES_RMO_CODE IN (1,7);

--Urban
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE QUES_RMO_CODE IN (2,3,5,9);

--Production
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE TO_NUMBER(SUBSTR(Q6_IND_CODE_CLASS_CODE,1,2)) BETWEEN 10 AND 35;

--Services
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE (TO_NUMBER(SUBSTR(Q6_IND_CODE_CLASS_CODE,1,2)) BETWEEN 05 AND 09) OR (TO_NUMBER(SUBSTR(Q6_IND_CODE_CLASS_CODE,1,2)) BETWEEN 35 AND 99);

--Cottage
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE UNIT_SIZE = 1 ;

--Micro
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE UNIT_SIZE = 2 ;

--Small
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE UNIT_SIZE = 3 ;

--Medium
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE UNIT_SIZE = 4 ;

--Large
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE UNIT_SIZE = 5 ;

--TIN Registered
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE Q25_IS_TIN_REGISTERED = 1 ;


--VAT Registered
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE Q26_IS_VAT_REGISTERED = 1 ;
