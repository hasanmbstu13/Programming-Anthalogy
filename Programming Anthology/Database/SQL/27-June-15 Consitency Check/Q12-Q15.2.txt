--Q4 = 2 And Q12 year of start is null
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE Q4_UNIT_TYPE = 2
AND Q12_YEAR_OF_START IS NULL;

--Q12 < 4
SELECT COUNT(*)
FROM BBSEC2013_REPORTS
WHERE Q4_UNIT_TYPE = 2
AND LENGTH(Q12_YEAR_OF_START) < 4;
