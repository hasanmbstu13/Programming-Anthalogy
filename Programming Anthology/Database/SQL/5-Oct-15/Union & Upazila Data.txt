(SELECT COUNT(*) FROM BBSEC2013_REPORTS REPORT_1 WHERE REPORT_1.ZILA_CODE = 50 AND REPORT_1.UPZILA_CODE = 15 AND REPORT.UNION_CODE = REPORT_1.UNION_CODE ) ESTABLISHMENT,
(SELECT SUM(TOTAL_PERSON_ENGAGED) FROM BBSEC2013_REPORTS  REPORT_1 WHERE REPORT_1.ZILA_CODE = 50 AND REPORT_1.UPZILA_CODE = 15 AND REPORT.UNION_CODE = REPORT_1.UNION_CODE ) TPE
FROM BBSEC2013_REPORTS REPORT
GROUP BY REPORT.UNION_CODE,REPORT.UNION_NAME
ORDER BY REPORT.UNION_NAME;
